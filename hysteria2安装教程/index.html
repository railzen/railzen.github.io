# 安装Hysteria2服务端

bash <(curl -fsSL https://get.hy2.sh/)
1

# cd到hysteria目录下

cd /etc/hysteria/
1

# 生成自签名证书

openssl req -x509 -nodes -newkey ec:<(openssl ecparam -name prime256v1) -keyout /etc/hysteria/server.key -out /etc/hysteria/server.crt -subj "/CN=apple.com" -days 36500 && sudo chown hysteria /etc/hysteria/server.key && sudo chown hysteria /etc/hysteria/server.crt
1

# 配置文件修改config.yaml

listen: :46153
 
# 以下 acme 和 tls 字段，二选一
# 有域名部署的选择 acme ，无域名的选择 tls
# 选择 acme，必须注释掉 tls，反之一样
 
#acme:
#  domains:
#    - ssharp.ccssharp.cc        # 域名
#  email: your@email.com   # 邮箱，格式正确即可
 
tls:
  cert: /etc/hysteria/server.crt
  key: /etc/hysteria/server.key
 
auth:
  type: password
  password: 0ac82ed2-33e2-4222-a178-cfea4886b041
 
masquerade:
  type: proxy
  proxy:
    url: https://apple.com # 伪装网站
    rewriteHost: true

# 启动服务

启动：systemctl start hysteria-server.service
重启：systemctl restart hysteria-server.service
开机自启：systemctl enable hysteria-server.service
查看状态：systemctl status hysteria-server.service
